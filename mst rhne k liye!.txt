<script>
const kissEmojis = ['💋', '😘', '❤️', '💖', '🥰', '🌹', '✨'];
let showerInterval;

function selectYes() {
document.getElementById('yesOverlay').classList.add('active');
createKissShower();
showerInterval = setInterval(createKissShower, 150);
}

function createKissShower() {
for(let i = 0; i < 6; i++) {
const kiss = document.createElement('div');
kiss.classList.add('kiss-emoji');
kiss.innerText = kissEmojis[Math.floor(Math.random() * kissEmojis.length)];
kiss.style.left = Math.random() * 100 + 'vw';
const duration = 1.5 + Math.random() * 2;
kiss.style.animationDuration = duration + 's';
kiss.style.fontSize = (20 + Math.random() * 40) + 'px';
document.body.appendChild(kiss);
setTimeout(() => { kiss.remove(); }, duration * 1000);
}
}

function selectNo() {
const card = document.getElementById('mainCard');
card.classList.add('shake');
setTimeout(() => {
card.classList.remove('shake');
document.getElementById('noOverlay').classList.add('active');
}, 400);
}

function resetPage() {
document.getElementById('noOverlay').classList.remove('active');
}
